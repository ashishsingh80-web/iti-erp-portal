function getSheet_(name) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name);
  if (!sheet) throw new Error('Sheet not found: ' + name);
  return sheet;
}

function getHeaders_(sheetName) {
  const sheet = getSheet_(sheetName);
  return sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
}

function getHeaderMapFromHeaders_(headers) {
  const map = {};
  headers.forEach((header, index) => {
    const key = String(header || '').trim();
    if (key) map[key] = index + 1;
  });
  return map;
}

function getRowsAsObjects_(sheetName) {
  const sheet = getSheet_(sheetName);
  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();

  if (lastRow < 2 || lastCol < 1) return [];

  const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
  const values = sheet.getRange(2, 1, lastRow - 1, lastCol).getValues();

  return values.map((row, idx) => {
    const obj = {};
    headers.forEach((h, i) => {
      obj[h] = row[i];
    });
    obj.__rowNumber = idx + 2;
    return obj;
  });
}

function getValidStudents_() {
  const cache = CacheService.getScriptCache();
  const cacheKey = getStudentCacheKey_();
  const cached = cache.get(cacheKey);

  if (cached) {
    return JSON.parse(cached);
  }

  const rows = getRowsAsObjects_(SHEETS.students).filter(r => {
    return String(r.Institute_ID || '').trim() !== '' &&
           String(r.Student_Name || '').trim() !== '' &&
           String(r.Trade || '').trim() !== '';
  });

  cache.put(cacheKey, JSON.stringify(rows), 120);
  return rows;
}

function getMissingStep_(r) {
  if (String(r.Admission_Form_Status || '').trim() !== 'Completed') return 'Admission Form';
  if (String(r.Documents_Verified_Status || '').trim() !== 'Verified') return 'Document Verification';
  if (String(r.Undertaking_Signed_Status || '').trim() === 'Pending' || String(r.Undertaking_Signed_Status || '').trim() === '') return 'Undertaking Signature';
  if (!String(r.PRN_Number || '').trim()) return 'PRN Pending';
  if (!String(r.SCVTUP_Registration_No || '').trim()) return 'SCVT Pending';
  return 'Review';
}

function getResponsibleTeam_(r) {
  if (String(r.Admission_Form_Status || '').trim() !== 'Completed') return 'Admission Staff';
  if (String(r.Documents_Verified_Status || '').trim() !== 'Verified') return 'Admission Desk';
  if (String(r.Undertaking_Signed_Status || '').trim() === 'Pending' || String(r.Undertaking_Signed_Status || '').trim() === '') return 'Admission Staff';
  if (!String(r.PRN_Number || '').trim() || !String(r.SCVTUP_Registration_No || '').trim()) return 'Scholarship/PRN Desk';
  return 'Admin';
}

function saveStudent(formData) {
  const missing = STUDENT_REQUIRED_FIELDS.filter(f => !String(formData[f] || '').trim());
  if (missing.length > 0) {
    throw new Error('Missing required fields: ' + missing.join(', '));
  }

  validateStudentMasterData_(formData);

  const lock = LockService.getScriptLock();
  lock.waitLock(30000);

  try {
    const sheet = getSheet_(SHEETS.students);
    const headers = getHeaders_(SHEETS.students);
    const headerMap = getHeaderMapFromHeaders_(headers);

    if (!headerMap['Institute_ID']) {
      throw new Error('Institute_ID header not found in STUDENTS_MASTER.');
    }

    const targetRow = getNextWritableRowByHeader_(sheet, headerMap, 'Institute_ID', 2);
    writeManualFieldsFast_(sheet, targetRow, headerMap, formData, STUDENT_MANUAL_FIELDS);

    SpreadsheetApp.flush();
    clearStudentCache_();

    const studentId = headerMap['Student_ID']
      ? sheet.getRange(targetRow, headerMap['Student_ID']).getDisplayValue()
      : '';

    return {
      success: true,
      row: targetRow,
      studentId: studentId
    };
  } finally {
    lock.releaseLock();
  }
}

function getNextWritableRowByHeader_(sheet, headerMap, headerName, startRow) {
  const colIndex = headerMap[headerName];
  if (!colIndex) throw new Error('Header not found: ' + headerName);

  const lastRow = sheet.getLastRow();
  if (lastRow < startRow) return startRow;

  const values = sheet.getRange(startRow, colIndex, lastRow - startRow + 1, 1).getDisplayValues().flat();

  for (let i = 0; i < values.length; i++) {
    if (!String(values[i] || '').trim()) {
      return startRow + i;
    }
  }

  return lastRow + 1;
}

function writeManualFieldsFast_(sheet, rowNumber, headerMap, formData, fields) {
  const updates = [];

  fields.forEach(field => {
    const colIndex = headerMap[field];
    if (!colIndex) return;
    updates.push({
      colIndex,
      value: normalizeStudentFieldValue_(field, formData[field])
    });
  });

  updates.sort((a, b) => a.colIndex - b.colIndex);

  let group = [];
  for (let i = 0; i < updates.length; i++) {
    const curr = updates[i];
    const prev = group[group.length - 1];

    if (!prev || curr.colIndex === prev.colIndex + 1) {
      group.push(curr);
    } else {
      flushWriteGroup_(sheet, rowNumber, group);
      group = [curr];
    }
  }

  if (group.length) flushWriteGroup_(sheet, rowNumber, group);
}

function flushWriteGroup_(sheet, rowNumber, group) {
  const startCol = group[0].colIndex;
  const values = [group.map(item => item.value)];
  sheet.getRange(rowNumber, startCol, 1, group.length).setValues(values);
}

function normalizeStudentFieldValue_(field, value) {
  if (value === null || value === undefined) return '';

  const dateFields = [
    'Agreement_Effective_Date',
    'Query_Submission_Date',
    'Scholarship_Approved_Date',
    'Scholarship_Credit_Date',
    'Undertaking_Generated_On',
    'Completion_Date'
  ];

  const numberFields = [
    'Scholarship_Credited_Amount',
    'Fees_If_Scholarship',
    'Fees_If_No_Scholarship',
    'Undertaking_Print_Count'
  ];

  if (typeof value === 'string') value = value.trim();

  if (dateFields.includes(field)) {
    if (!value) return '';
    const d = new Date(value);
    return isNaN(d.getTime()) ? value : d;
  }

  if (numberFields.includes(field)) {
    if (value === '') return '';
    return isNaN(Number(value)) ? value : Number(value);
  }

  return value;
}

function validateStudentMasterData_(formData) {
  const cache = CacheService.getScriptCache();
  const cacheKey = 'iti_master_validation_v1';
  let masterData = cache.get(cacheKey);

  if (masterData) {
    masterData = JSON.parse(masterData);
  } else {
    masterData = {
      institutes: getRowsAsObjects_(SHEETS.institutes)
        .map(r => String(r.Institute_ID || '').trim())
        .filter(Boolean),
      trades: getRowsAsObjects_(SHEETS.trades)
        .map(r => String(r.Trade || '').trim().toLowerCase())
        .filter(Boolean),
      agents: getRowsAsObjects_(SHEETS.agents)
        .map(r => String(r.Agent_ID || '').trim())
        .filter(Boolean)
    };
    cache.put(cacheKey, JSON.stringify(masterData), 300);
  }

  const instituteId = String(formData.Institute_ID || '').trim();
  const trade = String(formData.Trade || '').trim().toLowerCase();
  const agentId = String(formData.Agent_ID || '').trim();

  if (instituteId && masterData.institutes.indexOf(instituteId) === -1) {
    throw new Error('Invalid Institute_ID: ' + instituteId);
  }
  if (trade && masterData.trades.indexOf(trade) === -1) {
    throw new Error('Invalid Trade: ' + formData.Trade);
  }
  if (agentId && masterData.agents.indexOf(agentId) === -1) {
    throw new Error('Invalid Agent_ID: ' + agentId);
  }
}

function saveStudentDocuments(studentId, studentName, files) {
  try {
    const rootFolder = DriveApp.getFolderById(CONFIG_DOCS.rootFolderId);
    const folderName = `${studentName} - ${studentId}`;
    const studentFolder = getOrCreateSubfolder_(rootFolder, folderName);

    const uploaded = [];

    files.forEach(file => {
      if (!file || !file.fileName || !file.base64Data) return;

      const contentType = file.mimeType || 'application/octet-stream';
      const bytes = Utilities.base64Decode(file.base64Data);
      const blob = Utilities.newBlob(bytes, contentType, file.fileName);
      const createdFile = studentFolder.createFile(blob);

      uploaded.push({
        documentType: file.documentType,
        fileName: createdFile.getName(),
        url: createdFile.getUrl(),
        fileId: createdFile.getId()
      });
    });

    return {
      success: true,
      folderName,
      uploaded
    };
  } catch (err) {
    throw new Error('saveStudentDocuments failed: ' + err.message);
  }
}

function getOrCreateSubfolder_(parentFolder, folderName) {
  const folders = parentFolder.getFoldersByName(folderName);
  if (folders.hasNext()) return folders.next();
  return parentFolder.createFolder(folderName);
}

  function renderTable(containerId, headers, rows) {

    if (!container) return;

    const html = `
      <table>
        <thead>
          <tr>${headers.map(h => `<th>${escapeHtml(h)}</th>`).join('')}</tr>

            rows && rows.length
              ? rows.map(r => `<tr>${r.map(c => `<td>${escapeHtml(c ?? '')}</td>`).join('')}</tr>`).join('')

  function fillSelect(id, items, includeBlank, blankLabel) {

  async function saveAdmission() {
    if (isSavingAdmission) return;

    const resultBox = document.getElementById('admissionResult');
    const saveBtn = document.querySelector('.button-row button');

    try {
      isSavingAdmission = true;
      if (saveBtn) saveBtn.disabled = true;
      if (resultBox) resultBox.innerHTML = 'Saving...';

      const form = document.getElementById('admissionForm');
      const formData = Object.fromEntries(new FormData(form).entries());

      const requiredFields = ['Institute_ID', 'Student_Name', 'Trade', 'Session', 'Year', 'Mobile'];
      const missing = requiredFields.filter(f => !String(formData[f] || '').trim());

      if (missing.length > 0) {
        if (resultBox) resultBox.innerHTML = `Please fill required fields: ${missing.join(', ')}`;
        isSavingAdmission = false;
        if (saveBtn) saveBtn.disabled = false;
        return;
      }

      google.script.run
        .withSuccessHandler(async (res) => {
          try {
            const studentId = res.studentId;
            const studentName = formData.Student_Name || '';
            const files = await collectDocumentFiles();

            if (files.length > 0 && studentId && studentName) {
              google.script.run
                .withSuccessHandler((uploadRes) => {
                  if (resultBox) {
                    resultBox.innerHTML =
                      `Saved successfully. Student ID: ${escapeHtml(studentId)}<br>Documents uploaded: ${(uploadRes.uploaded || []).length}`;
                  }
                  form.reset();
                  loadStudents();
                  loadDashboard();
                })
                .withFailureHandler((err) => {
                  if (resultBox) {
                    resultBox.innerHTML = `Student saved, but document upload failed: ${escapeHtml(err.message || String(err))}`;
                  }
                  loadStudents();
                  loadDashboard();
                })
                .saveStudentDocuments(studentId, studentName, files);
            } else {
              if (resultBox) resultBox.innerHTML = `Saved successfully. Student ID: ${escapeHtml(studentId)}`;
              form.reset();
              loadStudents();
              loadDashboard();
            }
          } catch (e) {
            if (resultBox) resultBox.innerHTML = `Saved, but post-save step failed: ${escapeHtml(e.message || String(e))}`;
          } finally {
            isSavingAdmission = false;
            if (saveBtn) saveBtn.disabled = false;
          }
        })
        .withFailureHandler((err) => {
          if (resultBox) resultBox.innerHTML = `Save failed: ${escapeHtml(err.message || String(err))}`;
          isSavingAdmission = false;
          if (saveBtn) saveBtn.disabled = false;
        })
        .saveStudent(formData);

    } catch (e) {
      if (resultBox) resultBox.innerHTML = `Form error: ${escapeHtml(e.message || String(e))}`;
      isSavingAdmission = false;
      if (saveBtn) saveBtn.disabled = false;
    }
  }

  async function collectDocumentFiles() {
    const fileConfigs = [
      { inputId: 'Photo_File', documentType: 'Photo' },
      { inputId: 'Aadhaar_File', documentType: 'Aadhaar' },
      { inputId: 'Caste_File', documentType: 'Caste Certificate' },
      { inputId: 'Income_File', documentType: 'Income Certificate' }
    ];

    const results = [];

    for (const cfg of fileConfigs) {
      const input = document.getElementById(cfg.inputId);
      if (input && input.files && input.files[0]) {
        const fileObj = await fileToBase64Object(input.files[0], cfg.documentType);
        results.push(fileObj);
      }
    }

    return results;
  }

  function fileToBase64Object(file, documentType) {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();

      reader.onload = function(e) {
        const dataUrl = e.target.result;
        const base64Data = dataUrl.split(',')[1];

        resolve({
          documentType,
          fileName: file.name,
          mimeType: file.type,
          base64Data
        });
      };

      reader.onerror = reject;
      reader.readAsDataURL(file);
    });
  }

  function formatLabel(key) {
    return String(key)
      .replace(/([A-Z])/g, ' $1')
      .replace(/^./, s => s.toUpperCase());
  }

  function escapeHtml(str) {
    return String(str)
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/"/g, '&quot;')
      .replace(/'/g, '&#039;');
  }
